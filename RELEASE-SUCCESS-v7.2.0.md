### **Basic Usage**

```javascript
const BFX = require("@jcbit/bitfinex-api-node");

const bfx = new BFX({
  apiKey: "your-api-key",
  apiSecret: "your-api-secret",
  ws: { autoReconnect: true, seqAudit: false, transform: true },
});

const ws = bfx.ws(2);
ws.on("open", () => {
  ws.subscribeTicker("BTCUSD");
});
```
